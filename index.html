Flowchart

Dibawah ini sistem kerja dari Water Tank Level:

Contoh script

  1. Initial
    2. -
  2. Close
    3. -
  3. Timing
    4. Interfal 1*10ms(cycle)

SVLOW=@W_D10
SVHIGH=@W_D11

IF SVLOW>0 THEN '\\WPSTATUS,FL1,LP1\\ON
    IF SVHIGH>SVLOW AND @B_M0=1 THEN
        @B_M10=1
        @B_M16=1
        @B_M12=1
    ENDIF
ENDIF

IF @W_D20>SVHIGH THEN '\\WPSTATUS,FL1\\OFF
    @B_M10=0
    @B_M16=0
    IF @W_D20<SVLOW AND @B_M1=0 THEN
        @B_M10=1
        @B_M16=1
        @B_M12=1
    ENDIF
ENDIF

IF @W_D20>SVLOW THEN '\\CVSTATUS,FL2\\ON,OFF
    IF @B_M3=1 THEN
        @B_M11=0
        @B_M15=0
    ELSE
        IF @B_M2=1 THEN
            @B_M11=1
            @B_M15=1
        ENDIF
    ENDIF
ELSE
    @B_M11=0
    @B_M15=0
ENDIF

IF @B_M1=1 THEN
    @B_M10=0
    @B_M40=0
    @W_D20=0
    @B_M15=0
    @B_M16=0
    @B_M11=0
    @W_D31=0
    @B_M12=0
ENDIF

IF @W_D20<SVLOW THEN
    @W_D40=0
ELSE
    IF @W_D20<SVHIGH THEN
        @W_D40=1
    ELSE
        @W_D40=2
    ENDIF
ENDIF

    Interfal 100*10ms(cycle)

IF @B_M16=1 THEN
    @W_D20=@W_D20+5
ENDIF
IF @B_M11=1 THEN
    @W_D20=@W_D20-2
ENDIF
  4. Trigger Bit
    5. -
